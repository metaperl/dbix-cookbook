
=pod

=head1 NAME

DBIx::Cookbook::Recipe::Searching::fetch_all - recipe to fetch all records

=head1 DESCRIPTION

We want the equivalent of C<SELECT * FROM table>

=head1 RECIPES

=head2 DBIx::Class

    package DBIx::Cookbook::DBIC::Command::fetch_all;
    use Moose;
    extends qw(MooseX::App::Cmd::Command);
    
    has order_by => (
    		 traits => [qw(Getopt)],
    		 isa => "Str",
    		 is  => "rw",
    		);
    
    
    sub execute {
      my ($self, $opt, $args) = @_;
    
      my $where = {};
      my $attr = {};
    
      if (my $val = $opt->{order_by}) {
        $attr->{order_by} = $val;
      }
    
      my $rs = $self->app->schema->resultset('Actor')->search($where, $attr);
    
      while (my $row = $rs->next) {
        use Data::Dumper;
        my %data = $row->get_columns;
        warn Dumper(\%data);
        
      }
    }
    
    1;
